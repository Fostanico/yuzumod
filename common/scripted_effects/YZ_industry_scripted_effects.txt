yz_civilian_transfer = {            
     # 1. 直接给国家增加对应的地图外民工  #如果你想抄这段代码，我建议你看船坞转移的那部分
    add_offsite_building = {            #因为Gemini一开始给我写的是关于船坞的，民工
        type = industrial_complex       #和军工的是我改的，有些注释我删掉了，所以逻辑有些
        level = one_time_ompression_for_civilian_factories   #跳跃，但是船坞的那个我是
    }                                   #完整复制下来了——2026.4.13，23：40，伊尔诺

    # 2. 设置一个临时变量，作为“还需要移除的地图内民工数量” (倒扣法指标)
    set_variable = {
        yz_remaining_industrial_complex_to_remove = one_time_ompression_for_civilian_factories
    }

    # 3. 开始循环寻找并移除
    while_loop_effect = {
        # 循环继续的条件：
        limit = {
            # 条件A: 指标还没扣完 (> 0)
            check_variable = {
                var = ROOT.yz_remaining_industrial_complex_to_remove
                value = 0
                compare = greater_than
            }
        }

        # 随机挑选一个当前控制的、且民工数量大于0的省份
        random_controlled_state = {
            limit = {
                industrial_complex > 0
            }

            if = {
                # 情况1：该省份的民工数量 > 我们还需要移除的指标
                # 比如该省有 5 个，但我们只需要再拆 2 个了
                limit = {
                    check_variable = {
                        var = THIS.building_level@industrial_complex
                        value = ROOT.yz_remaining_industrial_complex_to_remove
                        compare = greater_than
                    }
                }
                # 那么只移除我们需要的数量
                remove_building = {
                    type = industrial_complex
                    level = ROOT.yz_remaining_industrial_complex_to_remove
                }
                set_variable = { ROOT.yz_remaining_industrial_complex_to_remove = 0 }
            }
            else = {
                # 情况2：该省份的民工数量 <= 我们还需要移除的指标
                # 比如我们需要再拆 5 个，但这个省只有 3 个
                
                # 先在指标中扣掉这 3 个（一定要在拆除之前扣，否则读不到数据）
                subtract_from_variable = {
                    var = ROOT.yz_remaining_industrial_complex_to_remove
                    value = THIS.building_level@industrial_complex
                }
                # 然后把这个省的 3 个民工全部拆除
                remove_building = {
                    type = industrial_complex
                    level = THIS.building_level@industrial_complex
                }
                # 此时该省民工变为0，在下一次 while 循环时，自然会被排除在寻找范围外，完全不需要设置 state_flag！
            }
        }
    }
}

     # 1. 直接给国家增加对应的地图外船坞

        yz_remaining_dockyards_to_remove = one_time_ompression_for_dockyard
    }

    # 3. 开始循环寻找并移除
    while_loop_effect = {
        # 循环继续的条件：
        limit = {
            # 条件A: 指标还没扣完 (> 0)
            check_variable = {
                var = yz_remaining_dockyards_to_remove
                value = 0
                compare = greater_than
            }
            # 条件B (安全锁): 全国范围内必须还有地图内船坞可以拆
            # 如果玩家输入的转移值比全国总船坞还多，没有这个条件游戏会陷入死循环卡死！
            any_controlled_state = {
                dockyard > 0
            }
        }

        # 随机挑选一个当前控制的、且船坞数量大于0的省份
        random_controlled_state = {
            limit = {
                dockyard > 0
            }

            if = {
                # 情况1：该省份的船坞数量 > 我们还需要移除的指标
                # 比如该省有 5 个，但我们只需要再拆 2 个了
                limit = {
                    check_variable = {
                        var = THIS.building_level@dockyard
                        value = ROOT.yz_remaining_dockyards_to_remove
                        compare = greater_than
                    }
                }
                # 那么只移除我们需要的数量
                remove_building = {
                    type = dockyard
                    level = ROOT.yz_remaining_dockyards_to_remove
                }
                # 指标自我扣减，归零（循环会在下个回合自动结束）
                #subtract_from_variable = {
                #    var = ROOT.yz_remaining_dockyards_to_remove
                #    value = ROOT.yz_remaining_dockyards_to_remove 
                #}
                set_variable = { ROOT.yz_remaining_dockyards_to_remove = 0 }
            }
            else = {
                # 情况2：该省份的船坞数量 <= 我们还需要移除的指标
                # 比如我们需要再拆 5 个，但这个省只有 3 个
                
                # 先在指标中扣掉这 3 个（一定要在拆除之前扣，否则读不到数据）
                subtract_from_variable = {
                    var = ROOT.yz_remaining_dockyards_to_remove
                    value = THIS.building_level@dockyard
                }
                # 然后把这个省的 3 个船坞全部拆除
                remove_building = {
                    type = dockyard
                    level = THIS.building_level@dockyard
                }
                # 此时该省船坞变为0，在下一次 while 循环时，自然会被排除在寻找范围外，完全不需要设置 state_flag！
            }
        }
    }
