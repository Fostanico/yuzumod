# 柚子核心系统的派生值统一入口。以下效果均要求国家 scope；调用方修改源变量后
# 应立即刷新，并由既定周/月入口兜底。不要为界面即时性恢复全局 on_daily 轮询。

# 根据当前工厂数与千人人力计算新作月收入 yz_new_works（0..2000000）。
# 两个输入必须先缩放再相乘，并在乘法前做饱和判断，避免定点数中间值溢出。
yz_recalculate_new_works_income = {
	set_temp_variable = { temp_yz_new_works_factory_factor = num_of_factories }
	multiply_temp_variable = { temp_yz_new_works_factory_factor = 0.01 }
	clamp_temp_variable = {
		var = temp_yz_new_works_factory_factor
		min = 0
	}

	set_temp_variable = { temp_yz_new_works_manpower_factor = max_manpower_k }
	multiply_temp_variable = { temp_yz_new_works_manpower_factor = 0.06 }
	clamp_temp_variable = {
		var = temp_yz_new_works_manpower_factor
		min = 0
	}

	set_variable = { yz_new_works = 0 }
	if = {
		limit = {
			check_variable = { temp_yz_new_works_factory_factor > 0 }
			check_variable = { temp_yz_new_works_manpower_factor > 0 }
		}
		if = {
			limit = {
				check_variable = { temp_yz_new_works_factory_factor < 1 }
			}
			multiply_temp_variable = {
				temp_yz_new_works_factory_factor = temp_yz_new_works_manpower_factor
			}
			clamp_temp_variable = {
				var = temp_yz_new_works_factory_factor
				min = 0
				max = 2000000
			}
			set_variable = { yz_new_works = temp_yz_new_works_factory_factor }
		}
		else = {
			set_temp_variable = { temp_yz_new_works_threshold = 2000000 }
			divide_temp_variable = {
				temp_yz_new_works_threshold = temp_yz_new_works_factory_factor
			}
			if = {
				limit = {
					check_variable = {
						var = temp_yz_new_works_manpower_factor
						value = temp_yz_new_works_threshold
						compare = greater_than
					}
				}
				set_variable = { yz_new_works = 2000000 }
			}
			else = {
				multiply_temp_variable = {
					temp_yz_new_works_factory_factor = temp_yz_new_works_manpower_factor
				}
				set_variable = { yz_new_works = temp_yz_new_works_factory_factor }
			}
		}
	}
}

# 重建财政 GUI 和月结算共用的国家级派生账本；本效果不修改真实余额 yz_money。
# 工厂、拨款、真咲、新作和长期投资任一来源改变后调用，月结算前必须再调用一次。
yz_recalculate_monthly_money_ledger = {
	clamp_variable = { var = YzMonthlyFund_var min = 0 max = 5 }
	clamp_variable = { var = YzEducation_fund_var min = 0 max = 5 }
	clamp_variable = { var = YzAdministration_fund_var min = 0 max = 5 }
	clamp_variable = { var = YzMilitary_fund_var min = 0 max = 5 }

	set_temp_variable = { temp_yz_money_factory_count = num_of_factories }
	clamp_temp_variable = { var = temp_yz_money_factory_count min = 0 }
	set_variable = { monthly_YZ_money_factory_income = 0 }
	if = {
		limit = { check_variable = { temp_yz_money_factory_count < 30 } }
		multiply_temp_variable = { temp_yz_money_factory_count = 3 }
		set_variable = { monthly_YZ_money_factory_income = temp_yz_money_factory_count }
	}
	else_if = {
		limit = { check_variable = { temp_yz_money_factory_count < 80 } }
		add_to_temp_variable = { temp_yz_money_factory_count = -30 }
		multiply_temp_variable = { temp_yz_money_factory_count = 2.5 }
		add_to_temp_variable = { temp_yz_money_factory_count = 90 }
		set_variable = { monthly_YZ_money_factory_income = temp_yz_money_factory_count }
	}
	else = {
		add_to_temp_variable = { temp_yz_money_factory_count = -80 }
		multiply_temp_variable = { temp_yz_money_factory_count = 1.5 }
		add_to_temp_variable = { temp_yz_money_factory_count = 215 }
		clamp_temp_variable = {
			var = temp_yz_money_factory_count
			min = 0
			max = 100000
		}
		set_variable = { monthly_YZ_money_factory_income = temp_yz_money_factory_count }
	}

	set_variable = { monthly_YZ_money_fund_income = YzMonthlyFund_var }
	multiply_variable = { monthly_YZ_money_fund_income = 150 }

	set_variable = { monthly_YZ_money_masaki_income = monly_YZ_masaki_money_visible }
	clamp_variable = {
		var = monthly_YZ_money_masaki_income
		min = 0
		max = 10000
	}
	if = {
		limit = { NOT = { has_variable = yz_new_works } }
		set_variable = { yz_new_works = 0 }
	}
	clamp_variable = { var = yz_new_works min = 0 max = 2000000 }

	set_variable = { monthly_YZ_money_income = monthly_YZ_money_factory_income }
	add_to_variable = { monthly_YZ_money_income = monthly_YZ_money_fund_income }
	add_to_variable = { monthly_YZ_money_income = monthly_YZ_money_masaki_income }
	add_to_variable = { monthly_YZ_money_income = yz_new_works }
	clamp_variable = { var = monthly_YZ_money_income min = 0 max = 2100000 }

	set_variable = { YzMoneyOutcome_var = YzEducation_fund_var }
	add_to_variable = { YzMoneyOutcome_var = YzAdministration_fund_var }
	add_to_variable = { YzMoneyOutcome_var = YzMilitary_fund_var }
	multiply_variable = { YzMoneyOutcome_var = 100 }

	set_variable = { monthly_YZ_money_industry_outcome = 0 }
	if = {
		limit = { has_country_flag = Yoshino_industry_long_term_invest_now_start }
		set_variable = { monthly_YZ_money_industry_outcome = basic_additional_yz_money_spend }
		clamp_variable = {
			var = monthly_YZ_money_industry_outcome
			min = 0
			max = 100000
		}
	}

	set_variable = { monthly_YZ_money_total_outcome = YzMoneyOutcome_var }
	add_to_variable = { monthly_YZ_money_total_outcome = monthly_YZ_money_industry_outcome }
	set_variable = { monly_YZ_money_visible = monthly_YZ_money_income }
	subtract_from_variable = { monly_YZ_money_visible = monthly_YZ_money_total_outcome }
}

# 计算下次月结算将增加的好感，输出 monly_num_YZ_love_depth。
# 该值是界面预测缓存；月结算前重算，周入口仅负责避免显示长期陈旧。
yz_recalculate_monthly_love_gain = {
	if = {
		limit = {
			has_variable = love_depth
			has_variable = num_YZ_general
		}
		clear_variable = monly_num_YZ_love_depth
		set_temp_variable = { temp_yz_love_factory_count = num_of_factories }
		multiply_temp_variable = { temp_yz_love_factory_count = 0.1 }
		round_temp_variable = temp_yz_love_factory_count
		clamp_temp_variable = {
			var = temp_yz_love_factory_count
			min = 0
		}
		set_variable = { monly_num_YZ_love_depth = num_YZ_general }
		add_to_variable = { monly_num_YZ_love_depth = temp_yz_love_factory_count }
		add_to_variable = { monly_num_YZ_love_depth = stability }
		round_variable = monly_num_YZ_love_depth
	}
}

# 重建柚子厨总量和可招募量。recruited_yzc_manpower 是已投入部队的人力，
# 既参与总量回填也参与最终可招募量修正；不要改成每日累加或清空。
yz_refresh_yzc_manpower_values = {
	if = {
		limit = {
			has_country_flag = qiyong_youzi_2
		}
		clamp_variable = {
			var = yzc_manpower_factor
			min = 0
			max = 1
		}
		set_variable = { yzc_manpower_k = 0 }
		set_temp_variable = { temp_yzc_unit_recruit = 0 }
		set_temp_variable = { temp_yzc_manpower = max_manpower_k }
		multiply_temp_variable = { temp_yzc_manpower = yzc_manpower_factor }
		add_to_variable = { yzc_manpower_k = temp_yzc_manpower }
		add_to_variable = { yzc_manpower_k = recruited_yzc_manpower }
		add_to_temp_variable = { temp_yzc_unit_recruit = yzc_manpower_k }
		set_temp_variable = { temp_yzc_zhandouli = yzc_zhandouli }
		add_to_temp_variable = { temp_yzc_zhandouli = -0.025 }
		multiply_temp_variable = { temp_yzc_unit_recruit = temp_yzc_zhandouli }
		set_variable = { yzc_unit_recruit = temp_yzc_unit_recruit }
		add_to_variable = {
			var = yzc_unit_recruit
			value = recruited_yzc_manpower
		}
	}
}

roka_koharu_check = {   #芦花小春触发事件判定
	if = {
		limit = {
			check_variable = {
				koharu_favor > 5
			}
			check_variable = {
				roka_favor > 5
			}
			check_variable = {
				roko_intensity > 2
			}
			has_country_flag = start_roko
		}
		country_event = RoKo_event.1
	}
	else_if = {
		limit = {
			check_variable = {
				koharu_favor > 4
			}
			check_variable = {
				roka_favor > 4
			}
			check_variable = {
				roko_intensity < roko_peace_factor
			}
			has_country_flag = start_roko
		}
		country_event = RoKo_event.2
	}
	else_if = {
		limit = {
			check_variable = {
				roka_favor > 8
			}
			has_country_flag = start_roko
		}
		country_event = RoKo_event.3
	}
	else_if = {
		limit = {
			check_variable = {
				koharu_favor > 8
			}
			has_country_flag = start_roko
		}
		country_event = RoKo_event.4
	}
	else = {

	}
}

turn_to_yzc = {	#变成柚子厨
	if = {
		limit = {
			has_equipment = {
				yz_works_equipment > 9
			}
			any_army_leader = {
				not = { has_trait = yzc_trait }
				yz_general = no
			}
		}
		random_army_leader = {
			limit = {
				not = { yz_general = yes }
			}
			add_unit_leader_trait = yzc_trait
		}
		add_equipment_to_stockpile = {
			type = yz_works_equipment
			amount = -10
			producer = JAP
		}
		custom_effect_tooltip = yzc_general_gain_tt
	}
}
