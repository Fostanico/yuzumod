# 玖玖瑠的国家级每周结算。这里只推进持久好感/研究进度；派生动态修正和完成事件
# 由统一 scripted effect 处理，所有离散写入点也调用同一入口，因此不需要 on_daily。
on_actions = {
	on_weekly = {
		effect = {
			if = {
				limit = { has_country_flag = get_kkr }
				add_to_variable = { kkr_favor = weekly_kkr_favor }
				kkr_refresh_favor_outputs = yes

				if = {
					limit = { has_country_flag = kkr_is_researching }
					add_to_variable = {
						kkr_research_progress = weekly_kkr_research_progress
					}
					clamp_variable = {
						var = kkr_research_progress
						min = 0
						max = 100
					}
					kkr_check_research_completion = yes
				}
			}
		}
	}
}
