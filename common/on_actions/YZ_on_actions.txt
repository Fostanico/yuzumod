# 柚子核心周期入口。HOI4 的周期 on_action 会按国家调用；月度系统必须保持玩家与
# 启用 flag 门槛。on_daily 仅保留需要逐日精度的生日计数，派生值走事件/周/月刷新。
on_actions = {
    on_weekly = {      #每周统一入口：防止被大清洗、低好感离开
        effect = {
            if = {
                limit = {
                    tag = SOV
                }
                every_army_leader = {
                    limit = {
                        yz_general = yes
                    }
                    set_character_flag = SOV_safe_from_purge_flag
                }
                every_navy_leader = {
                    limit = {
                        yz_general = yes
                    }
                    set_character_flag = SOV_safe_from_purge_flag
                }
                every_character = {
                    limit = {
                        is_yz_advisor = yes
                    }
                    set_character_flag = SOV_safe_from_purge_flag
                }
            }

            if = {
                limit = {
                    has_variable = love_depth
                    has_variable = num_YZ_general
                    check_variable = { love_depth < -50 }
                }
                random_list = {
                    50 = { add_to_variable = { love_depth = 10 } }
                }
                random_list = {
                    30 = {
                        random_army_leader = {
                            limit = {
                                yz_general = yes
                            }
                            remove_unit_leader_role = yes
                        }
                        country_event = start_event.3
                    }
                }
            }

            # 每周刷新玩家界面显示的“每月少女好感变化”预测。
            yz_recalculate_monthly_love_gain = yes
            yz_refresh_love_modifier_values = yes
            yz_refresh_yzc_manpower_values = yes
        }
    }

    on_startup = {
        effect = {
            # 仅在新开局判定一次：2000 年后且俄车战争已由千禧黎明历史文件建立时，
            # 记录当前存档使用千禧黎明规则，供学说与现代资源决议分流。
            if = {
                limit = {
                    has_start_date > 1999.12.31
                    SOV = { has_war_with = CHE }
                }
                set_global_flag = yz_md_startup_detected
            }

            set_victory_points = {
                province = 12097
                value = 30
            } #穗织
            set_province_name = {
                id = "12097"
                name = "穗织"
            }
            set_victory_points = {
                province = 12006
                value = 30
            } #御一夜
            set_province_name = {
                id = "12006"
                name = "御一夜"
            }
            set_victory_points = {
                province = 7102
                value = 30
            }
            set_province_name = {
                id = 7102
                name = "鹫逗"
            }
            set_variable = { date_counter = 0 }

        }
    }

    # 柚子币唯一月结算入口：其他 on_action 不得再次扣除部门或长期投资支出。
    on_monthly = {     #每月统一入口：商店限制、好感、工业资金与坏结局
        effect = {
            ROOT = {
                if = {
                    limit = {
                        is_ai = no
                        OR = {
                            has_country_flag = qiyong_youzi_2
                            has_country_flag = qiyong_youzi_3
                        }
                    }

                    set_variable = { yz_money_pp = 5 }
                    set_variable = { yz_money_cp = 5 }
                    set_variable = { yz_money_xp1 = 5 }
                    set_variable = { yz_money_xp2 = 5 }
                    set_variable = { yz_money_xp3 = 5 }
                    set_variable = { YzStore_5_var = 15 }
                    set_variable = { YzStore_6_var = 10 }
                    set_variable = { YzStore_8_var = 3 }
                    set_variable = { YzStore_10_var = 5 }
                    set_variable = { YzStore_11_var = 1 }

                    if = {
                        limit = {
                            has_variable = love_depth
                            has_variable = num_YZ_general
                        }
                        yz_recalculate_monthly_love_gain = yes
                        add_to_variable = { love_depth = monly_num_YZ_love_depth }
                        set_variable = { base_max_love = 100 }
                        add_to_variable = { base_max_love = marry_love_bonus }
                        clamp_variable = {		#范围
                            var = love_depth
                            min = -100
                            max = base_max_love
                        }
                        yz_refresh_love_modifier_values = yes
                    }

                    # 先以当月最新输入重建账本，再在临时余额中执行单次结算。
                    yz_recalculate_monthly_money_ledger = yes
                    set_temp_variable = { temp_yz_money_monthly_balance = yz_money }
                    add_to_temp_variable = {
                        temp_yz_money_monthly_balance = monthly_YZ_money_income
                    }
                    subtract_from_temp_variable = {
                        temp_yz_money_monthly_balance = YzMoneyOutcome_var
                    }

                    # 投资不足 flag 只描述本月结果，必须先清除旧月份状态。
                    clr_country_flag = monly_yz_money_spent_too_much
                    if = {
                        limit = {
                            has_country_flag = Yoshino_industry_long_term_invest_now_start
                        }
                        if = {
                            limit = {
                                check_variable = {
                                    var = temp_yz_money_monthly_balance
                                    value = monthly_YZ_money_industry_outcome
                                    compare = greater_than_or_equals
                                }
                            }
                            subtract_from_temp_variable = {
                                temp_yz_money_monthly_balance = monthly_YZ_money_industry_outcome
                            }
                        }
                        else = {
                            set_country_flag = monly_yz_money_spent_too_much
                        }
                    }

                    # 最后一次性写回并保护余额，避免中途负值被其他消费者观察到。
                    set_variable = { yz_money = temp_yz_money_monthly_balance }
                    clamp_variable = {		#范围
                        var = yz_money
                        min = 0
                        max = yz_money_max
                    }
                    set_temp_variable = { yz_money_temp_1 = 0 }
                    add_to_temp_variable = { yz_money_temp_1 = YzMonthlyFund_var }
                    multiply_temp_variable = { yz_money_temp_1 = -10 }
                    meta_effect = {		#自动比例扣除pp
                        text = {
                            add_political_power = [TEMP_POL]
                        }
                        TEMP_POL = "[?yz_money_temp_1|0]" # 调用的变量
                    }
                    round_variable = yz_money #取整

                    if = {
                        limit = {
                            check_variable = {
                                var = yz_money
                                value = 1
                                compare = less_than_or_equals
                            }
                        }
                        country_event = start_event.11
                    }

                    if = {
                        limit = {
                            has_variable = love_depth
                            has_variable = num_YZ_general
                            check_variable = { love_depth = -100 }
                        }
                        every_army_leader = {
                            limit = {
                                yz_general = yes
                            }
                            remove_unit_leader_role = yes
                        }
                        country_event = start_event.4
                        set_country_flag = yz_bad_end
                    }
                }
            }
        }
    }

    on_daily = {       #佐奈生日日期需要按日精确推进，但只为玩家国家写入变量
        effect = {
            if = {
                limit = { is_ai = no }
                add_to_variable = { date_counter = 1 }
                if = {
                    limit = { check_variable = { date_counter > 364 } }
                    add_to_variable = { date_counter = -365 }
                }
            }
        }
    }

    on_deployed_leader_defeated = {         #KR里搬过来的喵，KR的代码就是比蠢驴的代码好用喵 ——huhu 2026.7.13 21:26
        effect = {              #将领防抓小机制，但是直接范围覆盖全游戏了，原版将也不会被抓了
            log = "on_deployed_leader_defeated [GetLogThis] [GetLogRoot] [GetLogFrom]"
			#if = { #TODO: DECIDE WITH TEAM WHETHER GENERAL CAPTURES SHOULD BE ALLOWED
			#	limit = { can_be_captured = yes }
			#	capture_by = FROM
			#}
            if = {
                limit = { NOT = { has_trait = wounded } }
                add_timed_unit_leader_trait = {
                    trait = wounded
                    days = 90
                }
            }
        }
    }



    on_peace = {
        effect = {
            if = {
                limit = {
                    has_country_flag = tianyin_fighting
                }
                remove_advisor_role = {
                    character = Amane_advisor
                    slot = political_advisor
                }
                remove_ideas = yz_amane1
                set_country_flag = get_true_amane
                country_event = sairi_event.10
                clr_country_flag = tianyin_fighting
            }
        }
    }













    ##############################################################
    ####把新加的放在这上面

    on_capitulation = {
        effect = {
            if = {
                limit = {
                    has_variable = love_depth
                    has_variable = num_YZ_general
                    is_ai = no
                }
                add_to_variable = { love_depth = -50 }
                clamp_variable = {		#范围
                    var = love_depth
                    min = -100
                    max = base_max_love
                }
                yz_refresh_love_modifier_values = yes
            }
        }
    }

    on_uncapitulation = {
        effect = {
            if = {
                limit = {
                    has_variable = love_depth
                    has_variable = num_YZ_general
                    is_ai = no
                }
                add_to_variable = { love_depth = 50 }
                clamp_variable = {		#范围
                    var = love_depth
                    min = -100
                    max = base_max_love
                }
                yz_refresh_love_modifier_values = yes
            }
        }
    }


    on_nuke_drop = {
        effect = {
            set_variable = {
                global.nukecountry = ROOT
            }
            news_event = { id = nuke_dropped.12097 days = 1 }
            news_event = { id = nuke_dropped.7102 days = 1 }
        }
    }
}
