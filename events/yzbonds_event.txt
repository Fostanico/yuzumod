# 长期投资事件会改变 basic_additional_yz_money_spend 与对应 idea。
# 每个选项结束时重建显示账本；真实投资费用仍由唯一月结算入口扣除。
add_namespace = yzbonds_event
country_event = {
	id = yzbonds_event.1
	title = yzbonds_event.1.t
	desc = yzbonds_event.1.d
	is_triggered_only = yes
	option = {
		name = yzbonds_event.1.a
		custom_effect_tooltip = we_choice_to_contiune_yz_bonds
		trigger = {
			NOT = {
				has_country_flag = yz_debt_collapse
			}
		}
		add_to_variable = {
			var = yz_interest_rate_we_can_see
			value = 0.03
			tooltip = add_interest_rate_tooltip
		}
		add_to_variable = {
			var = yz_money
			value = 4000
		}
		add_to_variable = {
			var = bond_we_need_to_repay
			value = 1000
		}
		activate_mission = Yoshino_industry_debt_cycle
	}
	option = {
		name = yzbonds_event.1.b
		custom_effect_tooltip = we_choice_to_replay_yz_bonds
		trigger = {
			NOT = {
				has_country_flag = yz_debt_collapse
			}
			Yuzu_can_repay_bonds_by_yzmoney = yes
		}
		custom_effect_tooltip = now_we_repay_the_bonds_with_yzmoney_tooltip
		subtract_from_variable = {
			var = yz_money
			value = bond_we_need_to_repay
			#tooltip = repay_the_bonds_with_yzmoney_tooltip
		}
		set_country_flag = finished_repayment_of_bonds
		#activate_mission = Yoshino_industry_debt_cycle
	}
	option = {
		name = yzbonds_event.1.c
		custom_effect_tooltip = we_choic_to_replay_this_turn_yz_bonds
		trigger = {
			NOT = {
				has_country_flag = yz_debt_collapse
			}
			Yuzu_can_repay_bonds_by_yzmoney = yes
		}
		subtract_from_variable = {
			var = yz_money
			value = bond_we_need_to_repay
		}
		set_variable = {
			var = yz_interest_rate_we_can_see
			value = 0.03
		}
		activate_mission = Yoshino_industry_debt_cycle
	}
}
country_event = {
	id = yzbonds_event.2
	title = yzbonds_event.2.t
	desc = yzbonds_event.2.d
	is_triggered_only = yes
	option = {     #偿还部分债务
		name = yzbonds_event.2.a
		custom_effect_tooltip = we_replay_part_of_yz_bonds
		trigger = {
			check_variable = {
				var = yz_money
				value = 0
				compare = greater_than
			}
		}
		subtract_from_variable = {
			var = bond_we_need_to_repay
			value = yz_money
		}
		set_variable = {
			var = yz_money
			value = 0
		}
		add_to_variable = {
			var = yz_interest_rate_we_can_see
			value = 0.05
		}
	}
	option = {
		name = yzbonds_event.2.b
		custom_effect_tooltip = we_are_forced_to_replay_yz
		set_country_flag = finished_repayment_of_bonds
		add_timed_idea = {
			idea = now_we_repay_the_bonds_without_yzmoney
			days = 500
		}
		
	}
	option = {
		name = yzbonds_event.2.c
		custom_effect_tooltip = we_raise_the_cap_of_yz_bond
		add_to_variable = {
			var = yz_money_max
			value = 5000
			tooltip = yz_money_max_raise_tt
		}
		add_to_variable = {
			var = inter
			value = 0.05
			tooltip = add_interest_rate_tooltip
		}
		add_political_power = -100
	}
}
