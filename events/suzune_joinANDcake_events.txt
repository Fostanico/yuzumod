add_namespace = LY_event

country_event = { 		##
	id = LY_event.101
	title = LY_event.101.t
	desc = LY_event.101.d
	picture = GFX_LY_event

	is_triggered_only = yes

	option = {
		name = LY_event.101.a
		set_country_flag = have_suzune
		meta_effect = {
			text = {
				generate_character = {
					token_base = suzune
					name = "汐山凉音"
					gender = female
					nationality = JAP
					portraits = {
						civilian = {
							large = "gfx/leader/suzune.png"
							small = "gfx/leader/suzune2.png"
						}
					}
					allowed_civil_war = {
						is_ai = no
					}
					advisor = {
						slot = high_command
						cost = 100
						idea_token = suzune_command
						desc = suzune_army_desc
						ledger = army
						traits = { suzune_army  }
						ai_will_do = {
							factor = 0
						}
					}
					corps_commander = {
						desc = suzune_corps_desc
						traits={
							maid_suzune_trait
							organisational_leader
							adaptable
							logistics_wizard
						}
						skill=5
						attack_skill=3
						defense_skill=3
						planning_skill=3
						logistics_skill=9
					}
				}
			}
		}
		random_army_leader = {
			limit = {
				has_trait = maid_kanna_trait
				has_trait = maid_zaozi_trait
			}
			add_logistics = 1
		}
	}
}
